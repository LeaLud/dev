JavaScript

Es un lenguaje de programación interpretado, dialecto del estándar ECMAScript. Se define como orientado a objetos, basado en prototipos, imperativo, débilmente tipado y dinámico.

  • No requiere ningún tipo de instalación, ya que la mayoría de los navegadores web lo incorporan
  • Archivo vinculado ↗

    let varNum = 4

    1. Incrementando y reduciendo en 1 una variable numérica

      varNum ++ varNum --

    2. Operando con variables numérica y reasignándole el nuevo valor

      varNum += 5 varNum -= 5 varNum *= 5 varNum /= 5

  3. Boolean. Boleano, verdadero o falso

    let string = true | false function myFunction (a, b) { if (a < b) { return true; } else { return false; } } console.log(myFunction(6, 5));

  4. Null. Valor nulo

    let string = null

  5. Undefined. Valor no definido. viene por defecto

    let string = undefined

  6. Object.

    let string =

  7. Symbol.

    let string =

Arrays y Arrays anidados

  1. Definir un array

    let myArray = ["Leandro", 37, 1986] let myArray2 = ["Federico", 9, 1986] let myArray3 = [["Leandro", 37, 1986], myArray2] console.log(myArray3[1][2]) console.log(JSON.stringify(myArray3[1][2]))

  2. Modificar el valor de un array

    let myArray = ["Leandro", 37, 1986] myArray[1] = 38

  3. Agregar un valor al comienzo de un array

    let myArray = ["Leandro", 37, 1986] myArray.unshift("Enero")

  4. Agregar un valor al final de un array

    let myArray = ["Leandro", 37, 1986] myArray.push("Enero")

  5. Quitar el primer valor de un array

    let myArray = ["Leandro", 37, 1986] myArray.shift()

  6. Quitar el último valor de un array

    let myArray = ["Leandro", 37, 1986] myArray.pop()

Switch

function switchValue(value) { let myVar switch(value) { case 1: myVar = "Value 1" break; case 2: case 3: myVar = "Value 2 y 3" break; case 4: case 5: case 6: myVar = "Value 4, 5 y 6" break; default: myVar = "Valor por defecto si no se cumple ninguna otra condición" break; } return myVar } console.log(switchValue(5)); console.log(switchValue(2)); console.log(switchValue(3)); console.log(switchValue(4)); console.log(switchValue(1));
let myVar = "Valor 3" switch(myVar) { case "Valor 1": console.log("Se muestra el valor 1") break; case "Valor 2": console.log("Se muestra el valor 2") break; case "Valor 3": console.log("Se muestra el valor 3") break; case "Valor 4": console.log("Se muestra el valor 4") break; default: console.log("Valor por defecto si no se cumple ninguna otra condición") break; }

Objetos

  1. Definiendo un objeto e imprimiendo su contenido

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } console.log(varObject.Nombre) console.log(varObject.Edad) console.log(varObject["Oficio actual"]) console.log(varObject.Color[0]) console.log(varObject["Color"][2])
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } console.log(varObject[4][0]) console.log(varObject[1])

  2. Agregando un valor a un elemento del objeto

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } varObject.Color.push("Verde") console.log(varObject.Color[3])
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } varObject[4].push("Valor 6") console.log(varObject[4])

  3. Agregando un elemento al objeto

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } varObject.Altura = [1.79 + "cm", 1.8 + "cm"] varObject["Peso"] = 79 + "kg" console.log(varObject.Peso) console.log(varObject.Altura[1])
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } varObject[5] = ["Valor 6", "Valor 7"] varObject[6] = "Valor 8" console.log(varObject[5][1])

  4. Eliminando un elemento al objeto

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } delete varObject.Nombre delete varObject["Oficio actual"] delete varObject["Color"] console.log(varObject)
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } delete varObject[2] delete varObject[4][0] console.log(varObject)

  5. Cambiando el valor de un elemento

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } varObject[0] = "Federico" varObject.Color[2] = "Fucsia" console.log(varObject[0]) console.log(varObject.Color[2])
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } varObject[3] = "valor 88" console.log(varObject[4]) varObject[4][0] = 'valor 44' console.log(varObject[4][0])

  6. Reasignando valores a un elemento del objeto

    let varObject = { "Nombre": "Leandro", "Oficio actual": "Web developer", "Edad": 37 + " años", "Color": ["Blanco", "Azul", "Violeta"] } varObject.Color = ["Verde", "Dorado", "Plateado"] console.log(varObject.Color)
    let varObject = { 1: "valor 1", 2: "valor 2", 3: "valor 3", 4: ["valor 4", "Valor 5"] } console.log(varObject[4]) varObject[4] = ["Valor 1500", "Valor 85,6", 45] console.log(varObject[4])
